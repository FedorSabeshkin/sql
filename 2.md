# Второе задание 
1.
```sql
SELECT sq.*
FROM squads sq
WHERE sq.leader_id IS NULL;
```
2.
```sql
SELECT dw.*
FROM dwarves dw
WHERE dw.profession = "warrior"
AND dw.age > 150;
```
3.
```sql
SELECT  dw.*
FROM dwarves dw
INNER JOIN items it ON dw.dwarv_id = it.owner_id
WHERE it.type = "weapon"
DISTINCT dw.dwarv_id;
```
4.
```sql
SELECT  dw.*, COUNT(*)
FROM dwarves dw
INNER JOIN tasks ts ON ts.assigne_to = dw.dwarv_id
GROUP BY ts.status;
```
5. 
```sql
SELECT  ts.*, dw.*, sq.name, 
FROM dwarves dw
INNER JOIN tasks ts ON dw.dwarv_id = ts.assigned_to
INNER JOIN squads sq ON dw.squad_id = sq.squad_id
WHERE sq.name = "Guardians";
```
6.
```sql
SELECT rs.relationship, dw.id, dw.name, dw_f.id, dw_f.name
FROM relationships rs
INNER JOIN dwarfs dw ON dw.dwarf_id = rs.dwarf_id
INNER JOIN dwarfd dw_f ON dw_f.dwarf_id rs.related_to
WHERE rs.relationship = "rs.relationship";
```
